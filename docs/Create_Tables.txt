CREATE TABLE IF NOT EXISTS User (username VARCHAR(100) NOT NULL ,PRIMARY KEY(username),password INT NOT NULL, idCoupon INT NOT NULL)
|
CREATE TABLE IF NOT EXISTS Coupon (idCoupon INT(11) NOT NULL AUTO_INCREMENT,PRIMARY KEY(idCoupon),value INT NOT NULL)
|
CREATE TABLE IF NOT EXISTS Purchase (idPurchase INT(11) NOT NULL AUTO_INCREMENT,PRIMARY KEY(idPurchase),value INT NOT NULL)
|
CREATE TABLE IF NOT EXISTS Item (idItem INT(11) NOT NULL AUTO_INCREMENT,PRIMARY KEY(idItem),name VARCHAR(100) NOT NULL,value INT NOT NULL)
